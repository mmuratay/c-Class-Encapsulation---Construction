        {
            foreach (var item in kitaplar)
            {
                Console.WriteLine(item.Adi+""+item.ISDN+""+item.Yazar);
            }

            for (int i = 0; i < kitaplar.Count-1; i++)
            {
                Console.WriteLine(kitaplar[i].Adi + "" + kitaplar[i].ISDN + "" + kitaplar[i].Yazar);
            }
        }
